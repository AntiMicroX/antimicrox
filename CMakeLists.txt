            ADD_DEFINITIONS(-DLD_LIBRARY_PATH="${ANTILIB_PATH}:${LD_LIBRARY_PATH}")

            # those instructions have to be before add_executable() or add_library() is called, otherwise it has no effect

        endif()

            add_library( antilib
